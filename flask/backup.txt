FROM ubuntu:18.04

RUN mkdir /home/app && \
    apt-get update -y && \
    apt-get install -y python-pip python-dev && \
    python -m pip install flask && \
    python -m pip install SQLAlchemy && \
    python -m pip install Flask-SQLAlchemy

COPY . /app

WORKDIR /app

ENTRYPOINT [ "python" ]

CMD [ "main.py" ]
